``` r
pols_data=
  read_csv("./data/fivethirtyeight_datasets/pols-month.csv") %>%
  janitor::clean_names() %>%
  separate(mon,into = c("year","month","day")) %>%
  mutate(
  year = as.character(year),
  year = replace(year,year == "YES","TRUE"),
  year = replace(year, year == "NO","FALSE"),
  year = as.integer(year)
  ) %>%
  mutate(
  day = as.character(day),
  day = replace(day,day == "YES","TRUE"),
  day = replace(day, day == "NO","FALSE"),
  day = as.integer(day)
  ) %>%
  mutate(
  month = as.character(month),
  month = replace(month,month == "YES","TRUE"),
  month = replace(month, month == "NO","FALSE"),

  )
```
